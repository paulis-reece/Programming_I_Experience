void printSeats(char array[row][column]) {
  for (int r = 1; r <= row; r++) {
    for (int c = 1; c <= column; c++) {
      cout << '#';
    }
    cout << endl;
  }
  cout << endl;
  cout << "Back Row Seats" << endl;
}

  cout << "Back Row Seats" << endl;
}

void reservation(int Rows, int Cols, char array[row][column]) {
  if (array[Rows][Cols] != '*') {
    array[Rows][Cols] = '*';
  }
}

void mulipleReservation(int ROWS, int COLUMNS, int COLUMNS2, char array[row][column],
                        int &Revenue) {

      if (array[ROWS][COLUMNS2] != '*') {
        array[ROWS][COLUMNS2] = '*';
      }
    }
  }
}
